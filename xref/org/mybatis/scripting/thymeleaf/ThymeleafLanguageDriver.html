39  /**
40   * The {@code LanguageDriver} for integrating with Thymeleaf.
41   *
42   * @author Kazuki Shimizu
43   * @version 1.0.0
44   */
45  public class ThymeleafLanguageDriver implements LanguageDriver {
46  
47    private final ITemplateEngine templateEngine;
48  
49    /**
50     * Constructor for creating instance with default {@code TemplateEngine}.
51     */
52    public ThymeleafLanguageDriver() {
53      this.templateEngine = createDefaultTemplateEngine(ThymeleafLanguageDriverConfig.newInstance());
54    }
55  
56    /**
57     * Constructor for creating instance with user specified {@code Properties}.
58     *
59     * @param config
60     *          A user defined {@code ITemplateEngine} instance
61     */
62    public ThymeleafLanguageDriver(ThymeleafLanguageDriverConfig config) {
63      this.templateEngine = createDefaultTemplateEngine(config);
64      TemplateFilePathProvider.setLanguageDriverConfig(config);
65    }
66  
67    /**
68     * Constructor for creating instance with user defined {@code ITemplateEngine}.
69     *
70     * @param templateEngine
71     *          A user defined {@code ITemplateEngine} instance
72     */
73    public ThymeleafLanguageDriver(ITemplateEngine templateEngine) {
74      this.templateEngine = templateEngine;
75    }
76  
77    private ITemplateEngine createDefaultTemplateEngine(ThymeleafLanguageDriverConfig config) {
78      MyBatisDialectisDialect.html#MyBatisDialect">MyBatisDialect dialect = new MyBatisDialect(config.getDialect().getPrefix());
79      Likes likes = Likes.newBuilder().escapeChar(config.getDialect().getLikeEscapeChar())
80          .escapeClauseFormat(config.getDialect().getLikeEscapeClauseFormat())
81          .additionalEscapeTargetChars(config.getDialect().getLikeAdditionalEscapeTargetChars()).build();
82      dialect.setLikes(likes);
83  
84      // Create an ClassLoaderTemplateResolver instance
85      ClassLoaderTemplateResolver classLoaderTemplateResolver = new ClassLoaderTemplateResolver();
86      TemplateMode mode = config.isUse2way() ? TemplateMode.CSS : TemplateMode.TEXT;
87      classLoaderTemplateResolver.setOrder(1);
88      classLoaderTemplateResolver.setTemplateMode(mode);
89      classLoaderTemplateResolver
90          .setResolvablePatterns(Arrays.stream(config.getTemplateFile().getPatterns()).collect(Collectors.toSet()));
91      classLoaderTemplateResolver.setCharacterEncoding(config.getTemplateFile().getEncoding().name());
92      classLoaderTemplateResolver.setCacheable(config.getTemplateFile().isCacheEnabled());
93      classLoaderTemplateResolver.setCacheTTLMs(config.getTemplateFile().getCacheTtl());
94      classLoaderTemplateResolver.setPrefix(config.getTemplateFile().getBaseDir());
95  
96      // Create an StringTemplateResolver instance
97      StringTemplateResolver stringTemplateResolver = new StringTemplateResolver();
98      stringTemplateResolver.setOrder(2);
99      stringTemplateResolver.setTemplateMode(mode);
100 
101     // Create an TemplateEngine instance
102     TemplateEngine targetTemplateEngine = new TemplateEngine();
103     targetTemplateEngine.addTemplateResolver(classLoaderTemplateResolver);
104     targetTemplateEngine.addTemplateResolver(stringTemplateResolver);
105     targetTemplateEngine.addDialect(dialect);
106     targetTemplateEngine.setEngineContextFactory(
107         new MyBatisIntegratingEngineContextFactory(targetTemplateEngine.getEngineContextFactory()));
108 
109     // Create an TemplateEngineCustomizer instance and apply
110     final TemplateEngineCustomizer customizer = Optional.ofNullable(config.getCustomizer()).map(v -> {
111       try {
112         return v.getConstructor().newInstance();
113       } catch (InstantiationException | IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
114         throw new IllegalStateException("Cannot create an instance for class: " + v, e);
115       }
116     }).map(TemplateEngineCustomizer.class::cast).orElse(TemplateEngineCustomizer.BuiltIn.DO_NOTHING);
117     customizer.accept(targetTemplateEngine);
118 
119     return targetTemplateEngine;
120   }

122   /**
123    * {@inheritDoc}
124    */
125   @Override
126   public ParameterHandler createParameterHandler(MappedStatement mappedStatement, Object parameterObject,
127       BoundSql boundSql) {
128     return new DefaultParameterHandler(mappedStatement, parameterObject, boundSql);
129   }
130 
131   /**
132    * {@inheritDoc}
133    */
134   @Override
135   public SqlSource createSqlSource(Configuration configuration, XNode script, Class<?> parameterType) {
136     return createSqlSource(configuration, script.getNode().getTextContent(), parameterType);
137   }
138 
139   /**
140    * {@inheritDoc}
141    */
142   @Override
143   public SqlSource createSqlSource(Configuration configuration, String script, Class<?> parameterType) {
144     return new ThymeleafSqlSource(configuration, templateEngine, script.trim(), parameterType);
145   }
