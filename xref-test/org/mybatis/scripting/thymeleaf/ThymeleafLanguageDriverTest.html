63      Class.forName("org.hsqldb.jdbcDriver");

66      dataSource.setUser("sa");

68  
69      try (Connection conn = dataSource.getConnection()) {
70        try (Reader reader = Resources.getResourceAsReader("create-db.sql")) {
71          ScriptRunner runner = new ScriptRunner(conn);
72          runner.setLogWriter(null);
73          runner.setErrorLogWriter(null);
74          runner.runScript(reader);
75          conn.commit();
76        }
77      }
78  
79      TransactionFactory transactionFactory = new JdbcTransactionFactory();
80      Environment environment = new Environment("development", transactionFactory, dataSource);
81  
82      Configuration configuration = new Configuration(environment);
83      configuration.getLanguageRegistry().register(new ThymeleafLanguageDriver(new TemplateEngine()));
84      configuration.setDefaultScriptingLanguage(ThymeleafLanguageDriver.class);
85  
86      configuration.addMapper(NameMapper.class);
87      sqlSessionFactory = new SqlSessionFactoryBuilder().build(configuration);
88    }

110   @Test
111   void testDefaultConfig() {
112     DefaultTemplateEngineCustomizer.templateEngine = null;
113     Configuration configuration = new Configuration();
114     configuration.setDefaultScriptingLanguage(ThymeleafLanguageDriver.class);
115     new SqlSessionFactoryBuilder().build(configuration);
116 
117     TemplateEngine templateEngine = DefaultTemplateEngineCustomizer.templateEngine;
118     ClassLoaderTemplateResolver classLoaderTemplateResolver = TemplateEngineCustomizer
119         .extractTemplateResolver(templateEngine, ClassLoaderTemplateResolver.class)
120         .orElseGet(() -> Assertions.fail("Cannot a ClassLoaderTemplateResolver instance."));
121 
122     Assertions.assertEquals(TemplateMode.CSS, classLoaderTemplateResolver.getTemplateMode());
123     Assertions.assertTrue(classLoaderTemplateResolver.isCacheable());
124     Assertions.assertNull(classLoaderTemplateResolver.getCacheTTLMs());
125     Assertions.assertEquals("UTF-8", classLoaderTemplateResolver.getCharacterEncoding());
126     Assertions.assertEquals("", classLoaderTemplateResolver.getPrefix());
127     Assertions.assertEquals(new LinkedHashSet<>(Collections.singleton("*.sql")),
128         classLoaderTemplateResolver.getResolvablePatterns());
129 
130     StringTemplateResolver stringTemplateResolver = TemplateEngineCustomizer
131         .extractTemplateResolver(templateEngine, StringTemplateResolver.class)
132         .orElseGet(() -> Assertions.fail("Cannot a StringTemplateResolver instance."));
133     Assertions.assertEquals(TemplateMode.CSS, stringTemplateResolver.getTemplateMode());
134     Assertions.assertFalse(stringTemplateResolver.isCacheable());
135     Assertions.assertNull(stringTemplateResolver.getCacheTTLMs());
136 
137     templateEngine.getDialects().stream().filter(MyBatisDialect.class::isInstance).findFirst()
138         .map(MyBatisDialect.class::cast).ifPresent(v -> {
139           Assertions.assertEquals("mb", v.getPrefix());
140           Likes expression = (Likes) v.getExpressionObjectFactory().buildObject(null, null);
141           Assertions.assertEquals("ESCAPE '\\'", expression.escapeClause());
142         });
143   }

343       Assertions.assertEquals(
