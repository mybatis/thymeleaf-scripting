90    @Test
91    void testCollectionParamWithParamAnnotation() {
92      // collection type is array
93      try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
94        NameMapper mapper = sqlSession.getMapper(NameMapper.class);
95        List<Name> names = mapper.findByIds(1, 3, 4);
96        Assertions.assertEquals(3, names.size());
97        Assertions.assertEquals(1, names.get(0).getId());
98        Assertions.assertEquals(3, names.get(1).getId());
99        Assertions.assertEquals(4, names.get(2).getId());
100     }
101     // collection type is list
102     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
103       NameMapper mapper = sqlSession.getMapper(NameMapper.class);
104       List<Name> names = mapper.findByFirstNames(Arrays.asList("Wilma", "Pebbles", "Barney"));
105       Assertions.assertEquals(3, names.size());
106       Assertions.assertEquals(2, names.get(0).getId());
107       Assertions.assertEquals(3, names.get(1).getId());
108       Assertions.assertEquals(4, names.get(2).getId());
109     }
110     // collection is empty
111     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
112       NameMapper mapper = sqlSession.getMapper(NameMapper.class);
113       List<Name> names = mapper.findByFirstNames(Collections.emptyList());
114       Assertions.assertEquals(0, names.size());
115     }
116     // single value
117     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
118       NameMapper mapper = sqlSession.getMapper(NameMapper.class);
119       Name name = mapper.findByFirstNamesWithNotCollectionType("Wilma");
120       Assertions.assertEquals(2, name.getId());
121     }

345   @Test
346   void testBulkInsertAndSelect() {
347     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
348       PersonMapper mapper = sqlSession.getMapper(PersonMapper.class);
349       // Insert
350       List<Person> persons = new ArrayList<>();
351       {
352         Personmeleaf/integrationtest/domain/Person.html#Person">Person person = new Person();
353         person.setName("MyBatis 1");
354         List<Mail> mails = new ArrayList<>();
355         person.setMails(mails);
356         {
357           Mail/thymeleaf/integrationtest/domain/Mail.html#Mail">Mail mail = new Mail();
358           mail.setAddress("mybatis1.main@test.com");
359           mails.add(mail);
360         }
361         {
362           Mail/thymeleaf/integrationtest/domain/Mail.html#Mail">Mail mail = new Mail();
363           mail.setAddress("mybatis1.sub@test.com");
364           mails.add(mail);
365         }
366         persons.add(person);
367       }
368       {
369         Personmeleaf/integrationtest/domain/Person.html#Person">Person person = new Person();
370         person.setName("MyBatis 2");
371         List<Mail> mails = new ArrayList<>();
372         person.setMails(mails);
373         {
374           Mail/thymeleaf/integrationtest/domain/Mail.html#Mail">Mail mail = new Mail();
375           mail.setAddress("mybatis2.main@test.com");
376           mails.add(mail);
377         }
378         {
379           Mail/thymeleaf/integrationtest/domain/Mail.html#Mail">Mail mail = new Mail();
380           mail.setAddress("mybatis2.sub@test.com");
381           mails.add(mail);
382         }
383         persons.add(person);
384       }
385 
386       int maxMailId = Optional.ofNullable(mapper.getMaxMailId()).filter(x -> x != 0).orElse(-1);
387 
388       mapper.insertByBulk(persons);
389       mapper.insertMailsByBulk(persons);
390 
391       // Select
392       List<Person> loadedPersons = mapper.selectPersons(persons.get(0).getId(), persons.get(1).getId());
393       Assertions.assertEquals(2, loadedPersons.size());
394       {
395         Person person = loadedPersons.get(0);
396         Assertions.assertEquals(persons.get(0).getId(), person.getId());
397         Assertions.assertEquals("MyBatis 1", person.getName());
398         List<Mail> mails = person.getMails();
399         Assertions.assertEquals(2, mails.size());
400         {
401           Mail mail = mails.get(0);
402           Assertions.assertEquals(maxMailId + 1, mail.getId());
403           Assertions.assertEquals(persons.get(0).getId(), mail.getPersonId());
404           Assertions.assertEquals("mybatis1.main@test.com", mail.getAddress());
405         }
406         {
407           Mail mail = mails.get(1);
408           Assertions.assertEquals(maxMailId + 2, mail.getId());
409           Assertions.assertEquals(persons.get(0).getId(), mail.getPersonId());
410           Assertions.assertEquals("mybatis1.sub@test.com", mail.getAddress());
411         }
412       }
413       {
414         Person person = loadedPersons.get(1);
415         Assertions.assertEquals(persons.get(1).getId(), person.getId());
416         Assertions.assertEquals("MyBatis 2", person.getName());
417         List<Mail> mails = person.getMails();
418         Assertions.assertEquals(2, mails.size());
419         {
420           Mail mail = mails.get(0);
421           Assertions.assertEquals(maxMailId + 3, mail.getId());
422           Assertions.assertEquals(persons.get(1).getId(), mail.getPersonId());
423           Assertions.assertEquals("mybatis2.main@test.com", mail.getAddress());
424         }
425         {
426           Mail mail = mails.get(1);
427           Assertions.assertEquals(maxMailId + 4, mail.getId());
428           Assertions.assertEquals(persons.get(1).getId(), mail.getPersonId());
429           Assertions.assertEquals("mybatis2.sub@test.com", mail.getAddress());
430         }
431       }
432       // Select using list property
433       {
434         PersonMapper.Conditions conditions = new PersonMapper.Conditions();
435         conditions.setMails(Arrays.asList("mybatis1.main@test.com", "mybatis2.sub@test.com"));
436         List<Mail> mails = mapper.selectMailsByConditions(conditions);
437         Assertions.assertEquals(2, mails.size());
438         {
439           Mail mail = mails.get(0);
440           Assertions.assertEquals(maxMailId + 1, mail.getId());
441           Assertions.assertEquals(persons.get(0).getId(), mail.getPersonId());
442           Assertions.assertEquals("mybatis1.main@test.com", mail.getAddress());
443         }
444         {
445           Mail mail = mails.get(1);
446           Assertions.assertEquals(maxMailId + 4, mail.getId());
447           Assertions.assertEquals(persons.get(1).getId(), mail.getPersonId());
448           Assertions.assertEquals("mybatis2.sub@test.com", mail.getAddress());
449         }
450       }
451       // Select using list property on iteration object
452       {
453         PersonMapper.Conditions conditions1 = new PersonMapper.Conditions();
454         conditions1.setMails(Arrays.asList("mybatis1.main@test.com", "mybatis2.sub@test.com"));
455         PersonMapper.Conditions conditions2 = new PersonMapper.Conditions();
456         conditions2.setMails(Collections.singletonList("mybatis1.sub@test.com"));
457         List<Mail> mails = mapper.selectMailsByConditionsArray(Arrays.asList(conditions1, conditions2));
458         Assertions.assertEquals(3, mails.size());
459         {
460           Mail mail = mails.get(0);
461           Assertions.assertEquals(maxMailId + 1, mail.getId());
462           Assertions.assertEquals(persons.get(0).getId(), mail.getPersonId());
463           Assertions.assertEquals("mybatis1.main@test.com", mail.getAddress());
464         }
465         {
466           Mail mail = mails.get(1);
467           Assertions.assertEquals(maxMailId + 2, mail.getId());
468           Assertions.assertEquals(persons.get(0).getId(), mail.getPersonId());
469           Assertions.assertEquals("mybatis1.sub@test.com", mail.getAddress());
470         }
471         {
472           Mail mail = mails.get(2);
473           Assertions.assertEquals(maxMailId + 4, mail.getId());
474           Assertions.assertEquals(persons.get(1).getId(), mail.getPersonId());
475           Assertions.assertEquals("mybatis2.sub@test.com", mail.getAddress());
476         }
477       }
478     }
