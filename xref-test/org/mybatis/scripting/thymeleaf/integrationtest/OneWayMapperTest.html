50    @BeforeAll
51    static void setUp() throws Exception {
52      Class.forName("org.hsqldb.jdbcDriver");
53      JDBCDataSource dataSource = new JDBCDataSource();
54      dataSource.setUrl("jdbc:hsqldb:mem:db1");
55      dataSource.setUser("sa");
56      dataSource.setPassword("");
57  
58      try (Connection conn = dataSource.getConnection()) {
59        try (Reader reader = Resources.getResourceAsReader("create-db.sql")) {
60          ScriptRunner runner = new ScriptRunner(conn);
61          runner.setLogWriter(null);
62          runner.setErrorLogWriter(null);
63          runner.runScript(reader);
64          conn.commit();
65        }
66      }
67  
68      TransactionFactory transactionFactory = new JdbcTransactionFactory();
69      Environment environment = new Environment("development", transactionFactory, dataSource);
70  
71      Configuration configuration = new Configuration(environment);
72      configuration.setMapUnderscoreToCamelCase(true);
73      configuration.getLanguageRegistry()
74          .register(new ThymeleafLanguageDriver(ThymeleafLanguageDriverConfig.newInstance(c -> c.setUse2way(false))));
75      configuration.setDefaultScriptingLanguage(ThymeleafLanguageDriver.class);
76  
77      configuration.addMapper(OneWayNameMapper.class);
78      configuration.addMapper(OneWayPersonMapper.class);
79      sqlSessionFactory = new SqlSessionFactoryBuilder().build(configuration);
80    }

268   @Test
269   void testEscapeLikeWildcard() {
270     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
271       OneWayNameMapper mapper = sqlSession.getMapper(OneWayNameMapper.class);
272       NameParam param = new NameParam();
273       param.setFirstName("Be%");
274       List<Name> names = mapper.findByName(param);
275       Assertions.assertEquals(1, names.size());
276       Assertions.assertEquals(names.get(0).getId(), 6);
277       Assertions.assertEquals(names.get(0).getFirstName(), "Be%ty");
278       Assertions.assertEquals(names.get(0).getLastName(), "Ab_le");
279     }
280     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
281       OneWayNameMapper mapper = sqlSession.getMapper(OneWayNameMapper.class);
282       NameParam param = new NameParam();
283       param.setLastName("Ab_");
284       List<Name> names = mapper.findByName(param);
285       Assertions.assertEquals(1, names.size());
286       Assertions.assertEquals(names.get(0).getId(), 6);
287       Assertions.assertEquals(names.get(0).getFirstName(), "Be%ty");
288       Assertions.assertEquals(names.get(0).getLastName(), "Ab_le");
289     }
290     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
291       OneWayNameMapper mapper = sqlSession.getMapper(OneWayNameMapper.class);
292       NameParam param = new NameParam();
293       param.setFirstName("Be\\");
294       List<Name> names = mapper.findByName(param);
295       Assertions.assertEquals(1, names.size());
296       Assertions.assertEquals(names.get(0).getId(), 7);
297       Assertions.assertEquals(names.get(0).getFirstName(), "Be\\ty");
298       Assertions.assertEquals(names.get(0).getLastName(), "Abble");
299     }
300     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
301       OneWayNameMapper mapper = sqlSession.getMapper(OneWayNameMapper.class);
302       NameParam param = new NameParam();
303       param.setFirstName("");
304       List<Name> names = mapper.findByName(param);
305       Assertions.assertEquals(7, names.size());
306     }

309   @Test
310   void testNestedCollectionProperty() {
311     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
312       OneWayPersonMapper mapper = sqlSession.getMapper(OneWayPersonMapper.class);
313       OneWayPersonMapper.Conditions conditions1 = new OneWayPersonMapper.Conditions();
314       conditions1.setMails(Arrays.asList("mybatis1.main@test.net", "mybatis2.sub@test.net"));
315       OneWayPersonMapper.Conditions conditions2 = new OneWayPersonMapper.Conditions();
316       conditions2.setMails(Collections.singletonList("mybatis1.sub@test.net"));
317       List<Mail> mails = mapper.selectMailsByConditionsArray(Arrays.asList(conditions1, conditions2));
318       Assertions.assertEquals(3, mails.size());
319       {
320         Mail mail = mails.get(0);
321         Assertions.assertEquals(1000, mail.getId());
322         Assertions.assertEquals(100, mail.getPersonId());
323         Assertions.assertEquals("mybatis1.main@test.net", mail.getAddress());
324       }
325       {
326         Mail mail = mails.get(1);
327         Assertions.assertEquals(1001, mail.getId());
328         Assertions.assertEquals(100, mail.getPersonId());
329         Assertions.assertEquals("mybatis1.sub@test.net", mail.getAddress());
330       }
331       {
332         Mail mail = mails.get(2);
333         Assertions.assertEquals(1003, mail.getId());
334         Assertions.assertEquals(101, mail.getPersonId());
335         Assertions.assertEquals("mybatis2.sub@test.net", mail.getAddress());
336       }
337     }
